== gulp-asciidoctor

=== Usage

.Simple
----
gulp.pipe(asciidoctor({
    header_footer: false,
    attributes: ['showtitle']
    }));
----

.Other argments
----
gulp.pipe(asciidoctor({
    base_dir:'path or pwd',
    header_footer: false,
    safe:'secured',//
    doctype:'article',////book,inline
    header_footer:true, // true or false
    attributes: ['showtitle']
    }));
----


=== install

----
npm install gulp-asciidoctor --save
----
